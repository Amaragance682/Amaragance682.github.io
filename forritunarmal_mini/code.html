;; Notkun: (sum1 n)
;; Fyrir: n er heiltala, n>=0
;; Gildi: Summan 0+1+...+n
;; Use: (sum1 n)
;; Pre: n is an integer, n>=0
;; Value: The sum 0+1+...+n




(define (sum1 n)
  (define (sum1-helper n acc) ;; sum1-helper tekur n og summuna acc
    (if (= n 0) ;; thangad til ad vid erum farin fra n alveg til 0
        acc     ;; skilar loka summu
        (sum1-helper (- n 1) (+ acc n)))) ;; recursive kallid til ad baeta i summuna
  (sum1-helper n 0)) ;; byrjar forritid og prestillir summuna sem 0


(sum1 5)






;; Notkun: (sum2 i n)
;; Fyrir: i og n eru heiltölur, i <= n+1
;; Gildi: Summan i+(i+1)+...+n, summa þeirra
;; heiltalna k þannig að i <= k <= n.
;; Use: (sum2 i n)
;; Pre: i and n are integers, i <= n+1
;; Value: The sum i+(i+1)+...+n, the sum of the
;; integers k such that i <= k <= n.


(define (sum2 i n)
  (define (sum2-helper i n acc)
    (if (> i n)
        acc
        (sum2-helper (+ i 1) n (+ acc i 1))
      )
    ) (sum2-helper i n 0)
  )


(sum2 11 10)
(sum2 11 20)






;; Notkun: ((sum3 i) n)
;; Fyrir: i og n eru heiltölur, i <= n+1
;; Gildi: Summan i+(i+1)+...+n
;; Use: ((sum3 i) n)
;; Pre: i and n are integers, i <= n+1
;; Value: The sum i+(i+1)+...+n




(define (sum3 i)
  (lambda (n)
    (define (sum3-helper i acc)
      (if (> i n)
          acc
          (sum3-helper (+ i 1) (+ acc i 1))
          )
      ) (sum3-helper i 0)
    )
  )


((sum3 11) 10)
((sum3 11) 20)






;; Notkun: (reviota n)
;; Fyrir: n sem er heiltala
;; Gildi: Skilar Fylki með tölum [n, n-1, n-2, ... , 3, 2, 1]


;; Notkun: (reviota-helper i acc)
;; Fyrir: i sem er byrjunin af tölunum í fylkið og acc sem er listinn sem heldur utan um stökin
;; Gildi ef satt: skilar núverandi acc lista.
;; Gildi ef ósatt: kallar aftur á sig sjálft og setur (i + 1) í i parameterin og (i stakinu við acc listan) í acc parameterin




(define (reviota n)
  (define (reviota-helper i acc)
    (if (> i n) ;; ef i er stærra en n
        acc ;; skilar lista
        (reviota-helper (+ i 1) (cons i acc)) ;; kallar á sig sjálft aftur en bætir 1 við i, og i í acc listan, notar svo þessi nýju gildi sem parameters
    ))
   (reviota-helper 1 '() ) ;; við viljum byrja með i = 1 og acc sem tóman lista
  )


(reviota 100)
